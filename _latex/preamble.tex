% chapter の最初のページにスタイルを適用
% https://github.com/texjporg/uplatex/blob/master/ujclasses.dtx#L2895
% arg1: 対象のコマンド
% arg2: 置換前のコード
% arg3: 置換後のコード
% arg4: デバック用
% arg5: デバック用
\makeatletter
\patchcmd{\chapter}{\thispagestyle{jpl@in}}{\thispagestyle{normal}}{}{}
\makeatother

% 表目次と図目次を追加
\makeatletter
\renewcommand{\sphinxtableofcontents}{
  \pagenumbering{roman}
  \begingroup
    \parskip
    \z@skip
    \sphinxtableofcontentshook
    \tableofcontents
    \listoffigures
    \listoftables
  \endgroup
  \if@openright
    \cleardoublepage
  \else
    \clearpage
  \fi
  \pagenumbering{arabic}
}
\makeatother
